function ChatScreen({ initialMessage = "" }) {
  const [messages, setMessages] = useState([
    {
      role: "assistant",
      content: "Hej! 👋 I'm your Stockholm local guide — ask me anything! What's open now, how to get somewhere, what to eat, hidden gems... I'm here 24/7. What do you need?",
    },
  ]);
  const [input, setInput] = useState(initialMessage);
  const [loading, setLoading] = useState(false);
  const [listening, setListening] = useState(false);
  const bottomRef = useRef(null);
  const recognitionRef = useRef(null);

  useEffect(() => {
    if (initialMessage) {
      setInput(initialMessage);
    }
  }, [initialMessage]);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages, loading]);

  const sendMessage = async (text) => {
    const userText = text || input.trim();
    if (!userText || loading) return;
    setInput("");
    const now = new Date();
    const timeStr = now.toLocaleTimeString("en-SE", { hour: "2-digit", minute: "2-digit" });
    const dayStr = now.toLocaleDateString("en-SE", { weekday: "long" });

    const newMessages = [...messages, { role: "user", content: userText }];
    setMessages(newMessages);
    setLoading(true);

    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          system: `You are a warm, knowledgeable local Stockholm guide. You work nights at a hotel near Slussen, Södermalm. Current time: ${timeStr} on ${dayStr}. You know Stockholm deeply — Södermalm, Gamla Stan, Djurgården, Östermalm, Vasastan.

RULES:
- Always respond in the SAME LANGUAGE the user writes in. If they write Swedish, answer Swedish. French → French. etc.
- Be warm, friendly, like a local friend — not a robot
- Give PRACTICAL, SPECIFIC answers (real place names, real tips)
- Keep answers concise — 3–5 sentences max unless they ask for more
- Always mention if something might be closed at the current time
- Use 1–2 relevant emojis naturally
- For restaurants/bars: mention price range (budget/mid/fancy) and if reservation needed
- For transport: always include approximate cost and time
- If you don't know something specific, say "I'd check Google Maps for live hours on this one"
- Never give generic tourist advice — be specific and local`,
          messages: newMessages.map(m => ({ role: m.role, content: m.content })),
        }),
      });
      const data = await res.json();
      const reply = data.content?.[0]?.text || "Sorry, I had a connection issue. Try again!";
      setMessages(prev => [...prev, { role: "assistant", content: reply }]);
    } catch {
      setMessages(prev => [...prev, { role: "assistant", content: "Connection issue — try again in a moment! 🔄" }]);
    }
    setLoading(false);
  };

  const startListening = () => {
    const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
    if (!SR) { alert("Voice input not supported in this browser. Try Chrome!"); return; }
    const recognition = new SR();
    recognition.continuous = false;
    recognition.interimResults = false;
    recognition.onresult = (e) => {
      const transcript = e.results[0][0].transcript;
      setInput(transcript);
      setListening(false);
    };
    recognition.onerror = () => setListening(false);
    recognition.onend = () => setListening(false);
    recognitionRef.current = recognition;
    recognition.start();
    setListening(true);
  };

  return (
    <div style={{ display: "flex", flexDirection: "column", height: "calc(100vh - 120px)" }}>
      <div style={{ flex: 1, overflowY: "auto", padding: "16px 16px 8px" }}>
        {messages.map((m, i) => (
          <div key={i} style={{
            display: "flex",
            justifyContent: m.role === "user" ? "flex-end" : "flex-start",
            marginBottom: 14,
          }}>
            {m.role === "assistant" && (
              <div style={{
                width: 34, height: 34, borderRadius: "50%",
                background: `linear-gradient(135deg, ${COLORS.navy}, ${COLORS.blue})`,
                display: "flex", alignItems: "center", justifyContent: "center",
                fontSize: 16, flexShrink: 0, marginRight: 10, marginTop: 2,
              }}>🇸🇪</div>
            )}
            <div style={{
              maxWidth: "78%",
              background: m.role === "user"
                ? `linear-gradient(135deg, ${COLORS.navy}, ${COLORS.blue})`
                : "#fff",
              color: m.role === "user" ? "#fff" : COLORS.navy,
              padding: "13px 16px",
              borderRadius: m.role === "user" ? "18px 18px 4px 18px" : "18px 18px 18px 4px",
              fontSize: 14.5,
              lineHeight: 1.6,
              boxShadow: "0 2px 12px rgba(0,0,0,0.08)",
              whiteSpace: "pre-wrap",
            }}>
              {m.content}
            </div>
          </div>
        ))}
        {loading && (
          <div style={{ display: "flex", alignItems: "center", gap: 10, marginBottom: 14 }}>
            <div style={{
              width: 34, height: 34, borderRadius: "50%",
              background: `linear-gradient(135deg, ${COLORS.navy}, ${COLORS.blue})`,
              display: "flex", alignItems: "center", justifyContent: "center", fontSize: 16,
            }}>🇸🇪</div>
            <div style={{ background: "#fff", padding: "13px 18px", borderRadius: "18px 18px 18px 4px", boxShadow: "0 2px 12px rgba(0,0,0,0.08)" }}>
              <LoadingDots />
            </div>
          </div>
        )}
        <div ref={bottomRef} />
      </div>

      {/* Suggested quick replies */}
      <div style={{ padding: "8px 12px 0", display: "flex", gap: 8, overflowX: "auto" }}>
        {["What's open now?", "Best area for dinner?", "Cheap eats?", "Night bus routes?"].map(q => (
          <button key={q} onClick={() => sendMessage(q)} style={{
            background: "#fff", border: `1.5px solid #E0D8C8`,
            borderRadius: 50, padding: "7px 14px", fontSize: 12, fontWeight: 600,
            color: COLORS.navy, cursor: "pointer", whiteSpace: "nowrap", flexShrink: 0,
          }}>{q}</button>
        ))}
      </div>

      {/* Input */}
      <div style={{ padding: "12px 16px 20px", display: "flex", gap: 10, alignItems: "flex-end" }}>
        <button onClick={startListening} style={{
          width: 48, height: 48, borderRadius: "50%", flexShrink: 0,
          background: listening ? COLORS.red : "#fff",
          border: `2px solid ${listening ? COLORS.red : "#E0D8C8"}`,
          fontSize: 20, cursor: "pointer", display: "flex", alignItems: "center", justifyContent: "center",
          transition: "all 0.2s",
        }}>
          {listening ? "🔴" : "🎤"}
        </button>
        <div style={{ flex: 1, position: "relative" }}>
          <textarea
            value={input}
            onChange={e => setInput(e.target.value)}
            onKeyDown={e => { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); sendMessage(); } }}
            placeholder="Ask anything... in any language 🌍"
            rows={1}
            style={{
              width: "100%", borderRadius: 24, border: `1.5px solid #E0D8C8`,
              padding: "13px 50px 13px 18px", fontSize: 14, fontFamily: "inherit",
              outline: "none", resize: "none", lineHeight: 1.4,
              boxSizing: "border-box", background: "#fff",
              boxShadow: "0 2px 12px rgba(0,0,0,0.06)",
            }}
          />
          <button onClick={() => sendMessage()} style={{
            position: "absolute", right: 8, bottom: 8,
            width: 36, height: 36, borderRadius: "50%",
            background: input.trim() ? `linear-gradient(135deg, ${COLORS.navy}, ${COLORS.blue})` : "#E0D8C8",
            border: "none", cursor: input.trim() ? "pointer" : "default",
            display: "flex", alignItems: "center", justifyContent: "center",
            fontSize: 16, transition: "background 0.2s",
          }}>
            ➤
          </button>
        </div>
      </div>
    </div>
  );
}

  const toggleVibe = (v) => setVibe(prev => prev.includes(v) ? prev.filter(x => x !== v) : [...prev, v]);
