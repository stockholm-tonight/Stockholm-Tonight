function Chat({initMsg="",location}){
  const [msgs,setMsgs]=useState([{role:"assistant",content:"Hej! 👋 I'm your Stockholm local guide — ask me anything in any language! Food, transport, hidden gems, what's open now... I'm here 24/7!"}]);
  const [input,setInput]=useState(initMsg);
  const [loading,setLoading]=useState(false);
  const bottomRef=useRef(null);

  useEffect(()=>{bottomRef.current?.scrollIntoView({behavior:"smooth"});},[msgs,loading]);

  const send=async(txt)=>{
    const text=txt||input.trim();
    if(!text||loading)return;
    setInput("");
    const now=new Date();
    const time=now.toLocaleTimeString("en-SE",{hour:"2-digit",minute:"2-digit"});
    const day=now.toLocaleDateString("en-SE",{weekday:"long"});
    const locStr=location?`User's current location: ${location}.`:"User is near Slussen, Södermalm.";
    const newMsgs=[...msgs,{role:"user",content:text}];
    setMsgs(newMsgs);
    setLoading(true);
    try{
      const res=await fetch("/.netlify/functions/chat",{
        method:"POST",
        headers:{"Content-Type":"application/json"},
        body:JSON.stringify({
          messages:newMsgs.map(m=>({role:m.role,content:m.content})),
          location:locStr
        })
      });
      const data=await res.json();
      if(data.error){
        setMsgs(p=>[...p,{role:"assistant",content:`⚠️ API Error: ${data.error.message}. Please check your API key in settings.`}]);
      } else {
        const reply=data.content?.[0]?.text||"Sorry, connection issue. Try again!";
        setMsgs(p=>[...p,{role:"assistant",content:reply}]);
      }
    }catch(e){
      setMsgs(p=>[...p,{role:"assistant",content:"Connection issue — check your internet and try again! 🔄"}]);
    }
    setLoading(false);
  };

  const mic=()=>{
    const SR=window.SpeechRecognition||window.webkitSpeechRecognition;
    if(!SR){alert("Voice input works in Chrome browser!");return;}
    const r=new SR();
    r.onresult=e=>setInput(e.results[0][0].transcript);
    r.start();
  };

  return <div style={{display:"flex",flexDirection:"column",height:"calc(100vh - 120px)"}}>
    <div style={{flex:1,overflowY:"auto",padding:"16px 16px 8px"}}>
      {msgs.map((m,i)=><div key={i} className="fade-in" style={{display:"flex",justifyContent:m.role==="user"?"flex-end":"flex-start",marginBottom:14}}>
        {m.role==="assistant"&&<div style={{width:34,height:34,borderRadius:"50%",background:`linear-gradient(135deg,${C.navy},${C.blue})`,display:"flex",alignItems:"center",justifyContent:"center",fontSize:16,flexShrink:0,marginRight:10,marginTop:2}}>🇸🇪</div>}
        <div style={{maxWidth:"78%",background:m.role==="user"?`linear-gradient(135deg,${C.navy},${C.blue})`:"#fff",color:m.role==="user"?"#fff":C.navy,padding:"13px 16px",borderRadius:m.role==="user"?"18px 18px 4px 18px":"18px 18px 18px 4px",fontSize:14.5,lineHeight:1.6,boxShadow:"0 2px 12px rgba(0,0,0,0.08)",whiteSpace:"pre-wrap"}}>{m.content}</div>
      </div>)}
      {loading&&<div style={{display:"flex",alignItems:"center",gap:10,marginBottom:14}}>
        <div style={{width:34,height:34,borderRadius:"50%",background:`linear-gradient(135deg,${C.navy},${C.blue})`,display:"flex",alignItems:"center",justifyContent:"center",fontSize:16}}>🇸🇪</div>
        <div style={{background:"#fff",padding:"13px 18px",borderRadius:"18px 18px 18px 4px",boxShadow:"0 2px 12px rgba(0,0,0,0.08)"}}><Dots/></div>
      </div>}
      <div ref={bottomRef}/>
    </div>
    <div style={{padding:"8px 12px 0",display:"flex",gap:8,overflowX:"auto"}}>
      {["What's open now?","Best area for dinner?","Night bus routes?","Free museums?"].map(q=>
        <button key={q} onClick={()=>send(q)} style={{background:"#fff",border:"1.5px solid #E0D8C8",borderRadius:50,padding:"7px 14px",fontSize:12,fontWeight:600,color:C.navy,cursor:"pointer",whiteSpace:"nowrap",flexShrink:0}}>{q}</button>
      )}
    </div>
    <div style={{padding:"12px 16px 20px",display:"flex",gap:10,alignItems:"flex-end"}}>
      <button onClick={mic} style={{width:48,height:48,borderRadius:"50%",flexShrink:0,background:"#fff",border:"2px solid #E0D8C8",fontSize:20,cursor:"pointer",display:"flex",alignItems:"center",justifyContent:"center"}}>🎤</button>
      <div style={{flex:1,position:"relative"}}>
        <textarea value={input} onChange={e=>setInput(e.target.value)} onKeyDown={e=>{if(e.key==="Enter"&&!e.shiftKey){e.preventDefault();send();}}} placeholder="Ask anything... in any language 🌍" rows={1} style={{width:"100%",borderRadius:24,border:"1.5px solid #E0D8C8",padding:"13px 50px 13px 18px",fontSize:14,fontFamily:"inherit",resize:"none",lineHeight:1.4,boxSizing:"border-box",background:"#fff",boxShadow:"0 2px 12px rgba(0,0,0,0.06)"}}/>
        <button onClick={()=>send()} style={{position:"absolute",right:8,bottom:8,width:36,height:36,borderRadius:"50%",background:input.trim()?`linear-gradient(135deg,${C.navy},${C.blue})`:"#E0D8C8",border:"none",cursor:"pointer",display:"flex",alignItems:"center",justifyContent:"center",fontSize:16}}>➤</button>
      </div>
    </div>
  </div>;
}

function App(){
  const [screen,setScreen]=useState("home");
  const [chatInit,setChatInit]=useState("");
  const [large,setLarge]=useState(false);

  const [location,setLocation]=useState("");

  // Get user location on load
  useEffect(()=>{
    if(navigator.geolocation){
      navigator.geolocation.getCurrentPosition(
        pos=>{
          const {latitude:lat,longitude:lng}=pos.coords;
          fetch(`https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lng}&format=json`)
            .then(r=>r.json())
            .then(d=>{
              const addr=d.address;
              const loc=addr.neighbourhood||addr.suburb||addr.city_district||addr.city||"Stockholm";
              setLocation(loc+", Stockholm");
            })
            .catch(()=>setLocation("Stockholm"));
        },
        ()=>setLocation("")
      );
    }
  },[]);



  const nav=(s,msg="")=>{
    if(msg){setChatInit(msg);setScreen("chat");}
    else{setScreen(s);if(s!=="chat")setChatInit("");}
    window.scrollTo(0,0);
  };

  const navItems=[
    {id:"home",emoji:"🏠",label:"Home"},
    {id:"chat",emoji:"💬",label:"Ask AI"},
    {id:"links",emoji:"🔗",label:"Links"},
    {id:"videos",emoji:"▶️",label:"Videos"},
    {id:"emergency",emoji:"🆘",label:"Help"},
  ];

  const titles={chat:"Ask Local AI 💬",transport:"Getting Around 🚇",food:"Food & Drinks 🍕",todo:"Things To Do 📍",gems:"Hidden Gems 💎",emergency:"Emergency Help 🆘",itinerary:"Build My Day 🗓️",links:"Useful Links 🔗",videos:"Video Guides ▶️"};

  return <div style={{fontFamily:"'Segoe UI',system-ui,sans-serif",background:"#FAF8F4",minHeight:"100vh",fontSize:large?"110%":"100%"}}>
    {screen!=="home"&&<div style={{background:"#fff",padding:"16px 20px",display:"flex",alignItems:"center",gap:14,borderBottom:"1px solid #F0E8D8",position:"sticky",top:0,zIndex:10,boxShadow:"0 2px 12px rgba(0,0,0,0.06)"}}>
      <button onClick={()=>setScreen("home")} style={{background:"#F5F0E8",border:"none",borderRadius:"50%",width:38,height:38,fontSize:18,cursor:"pointer",display:"flex",alignItems:"center",justifyContent:"center"}}>←</button>
      <div style={{fontSize:16,fontWeight:800,color:C.navy,flex:1}}>{titles[screen]}</div>
      <button onClick={()=>setLarge(p=>!p)} style={{background:large?C.gold:"#F5F0E8",border:"none",borderRadius:50,padding:"6px 12px",fontSize:12,fontWeight:700,cursor:"pointer",color:C.navy}}>{large?"A-":"A+"}</button>
    </div>}

    {screen==="home"&&<Home onNav={nav} location={location}/>}
    {screen==="chat"&&<Chat key={chatInit} initMsg={chatInit} location={location}/>}
    {screen==="transport"&&<Transport/>}
    {screen==="food"&&<Food onNav={nav}/>}
    {screen==="todo"&&<ToDo onNav={nav}/>}
    {screen==="gems"&&<Gems/>}
    {screen==="emergency"&&<Emergency/>}
    {screen==="itinerary"&&<Itinerary onNav={nav}/>}
    {screen==="links"&&<LinksScreen/>}
    {screen==="videos"&&<VideosScreen/>}

    <div style={{position:"fixed",bottom:0,left:"50%",transform:"translateX(-50%)",width:"100%",maxWidth:480,background:"#fff",borderTop:"1px solid #F0E8D8",display:"flex",padding:"8px 0 12px",boxShadow:"0 -4px 20px rgba(0,0,0,0.08)",zIndex:100}}>
      {navItems.map(item=><button key={item.id} onClick={()=>nav(item.id)} style={{flex:1,background:"none",border:"none",cursor:"pointer",display:"flex",flexDirection:"column",alignItems:"center",gap:4,padding:"6px 4px"}}>
        <span style={{fontSize:22,filter:screen===item.id?"none":"grayscale(0.3) opacity(0.6)"}}>{item.emoji}</span>
        <span style={{fontSize:10,fontWeight:700,color:screen===item.id?C.navy:C.muted,letterSpacing:0.5}}>{item.label}</span>
        {screen===item.id&&<div style={{width:20,height:3,background:C.gold,borderRadius:2}}/>}
      </button>)}
    </div>
  </div>;
}
